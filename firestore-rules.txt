// Firebase Firestore Security Rules for Bookings
// Go to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to create and read their own bookings
    match /bookings/{bookingId} {
      // Allow anyone authenticated to create bookings
      allow create: if request.auth != null;
      
      // Allow users to read their own bookings or admins to read all
      allow read: if request.auth != null && 
        (request.auth.uid == resource.data.userId || 
         request.auth.token.admin == true);
      
      // Allow users to update/delete their own bookings or admins to manage all
      allow update, delete: if request.auth != null && 
        (request.auth.uid == resource.data.userId || 
         request.auth.token.admin == true);
    }

    // TEMPORARY: Allow all operations for testing (REMOVE AFTER TESTING)
    match /bookings/{bookingId} {
      allow read, write: if request.auth != null;
    }

    // User profiles collection (for future use)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
